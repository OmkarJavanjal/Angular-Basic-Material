    function Remove(button) {
        //Determine the reference of the Row using the Button.
        var row = button.parentNode.parentNode;
        var name = row.getElementsByTagName("TD")[0].innerHTML;
        if (confirm("Do you want to delete: " + name)) {

            //Get the reference of the Table.
            var table = document.getElementById("tblCustomers");

            //Delete the Table row using it's Index.
            table.deleteRow(row.rowIndex);
        }
    };

        [WebMethod]
        public static List<ListItem> GetCustomers()
        {
            string query = "Select top 500 ID,Name from Updated_Project_Data Group BY ID,Name";
            string constr = "Data Source=aimlproject.database.windows.net;Initial Catalog=AIML;User ID=AIMLAdmin;Password=Admin@2018";
            using (SqlConnection con = new SqlConnection(constr))
            {
                using (SqlCommand cmd = new SqlCommand(query))
                {
                    List<ListItem> customers = new List<ListItem>();
                    cmd.CommandType = CommandType.Text;
                    cmd.Connection = con;
                    con.Open();
                    using (SqlDataReader sdr = cmd.ExecuteReader())
                    {
                        while (sdr.Read())
                        {
                            customers.Add(new ListItem
                            {
                                Value = sdr["ID"].ToString(),
                                Text = sdr["Name"].ToString()
                            });
                        }
                    }
                    con.Close();

                    return customers;

                }
            }
        }


        [WebMethod]
        public static List<ListItem> GetSkill(string ID)
        {
            int IDi = Convert.ToInt32(ID);
            string query = "Select Skill from [dbo].[Updated_Project_Data] where ID = " + IDi;
            string constr = "Data Source=aimlproject.database.windows.net;Initial Catalog=AIML;User ID=AIMLAdmin;Password=Admin@2018";

            using (SqlConnection con = new SqlConnection(constr))
            {
                using (SqlCommand cmd = new SqlCommand(query,con))
                {
                    List<ListItem> Skills = new List<ListItem>();
                    SqlDataAdapter sda = new SqlDataAdapter(cmd);

                    DataTable dt = new DataTable();
                    sda.Fill(dt);
                    foreach (DataRow item in dt.Rows)
                    {
                        Skills.Add(new ListItem
                        {
                            Value = item["Skill"].ToString(),
                            Text = item["Skill"].ToString()
                        });
                    }

                    return Skills;

                }
            }
